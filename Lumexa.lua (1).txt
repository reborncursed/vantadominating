--[[
    Lumexa - Combined Single-File Build
    Auto-generated by concatenating all 11 modules in dependency order
    (Config, Logger, Utilities, Status, Scanner, Webhook, SeaManager,
    BossManager, ServerHop, UI, Main) into one script, for environments
    where a single loadstring/script is required instead of a
    ModuleScript folder hierarchy.

    Each module below is unchanged except:
      - require(script.Parent.X) lines are removed — X is already
        an in-scope local table by the time it is needed, since
        modules are ordered by dependency.
      - Each module's trailing `return X` is removed (only Main's
        final return survives, since a return must be the last
        statement of the whole chunk).
      - The `script.Parent:FindFirstChild("_NoAutoStart")` guard in
        Main is removed — there is no module hierarchy to check
        here, so the framework just starts unconditionally.
]]

--- Marks the framework as running/stopped and tracks elapsed time
--- across start/stop cycles. Lives here (not in Main) so any module
--- that already requires Config — including UI — can read status
--- and uptime without creating a circular require on Main.
function Config.SetRunning(isRunning)
    local now = os.clock()
    if isRunning and not Config._runtime.Started then
        Config._runtime.Started = true
        Config._runtime.StartTime = now
    elseif not isRunning and Config._runtime.Started then
        Config._runtime.AccumulatedUptime = Config._runtime.AccumulatedUptime
            + (now - (Config._runtime.StartTime or now))
        Config._runtime.Started = false
        Config._runtime.StartTime = nil
    end
end

--- Total seconds the framework has been running, across start/stop
--- cycles. Freezes while stopped rather than continuing to climb.
function Config.GetUptime()
    if Config._runtime.Started and Config._runtime.StartTime then
        return Config._runtime.AccumulatedUptime + (os.clock() - Config._runtime.StartTime)
    end
    return Config._runtime.AccumulatedUptime
end

function Config.GetStatus()
    return Config._runtime.Started and "Running" or "Stopped"
end

--- Safely set a nested config value using dot path, e.g. "Scanner.Interval"
function Config.Set(path, value)
    local keys = {}
    for key in string.gmatch(path, "[^%.]+") do
        table.insert(keys, key)
    end

    local node = Config
    for i = 1, #keys - 1 do
        node = node[keys[i]]
        if type(node) ~= "table" then
            return false, "Invalid config path: " .. path
        end
    end

    local lastKey = keys[#keys]
    if node[lastKey] == nil and node ~= Config then
        return false, "Unknown config key: " .. path
    end

    node[lastKey] = value
    return true
end

--- Safely get a nested config value using dot path
function Config.Get(path)
    local keys = {}
    for key in string.gmatch(path, "[^%.]+") do
        table.insert(keys, key)
    end

    local node = Config
    for _, key in ipairs(keys) do
        if type(node) ~= "table" then
            return nil
        end
        node = node[key]
    end

    return node
end

--[[
    Lumexa - Logger Module
    Centralized logging. Every write is pcall-protected so a logging
    failure never crashes the caller.
]]


local Logger = {}

Logger._buffer = {} -- in-memory ring buffer of log lines
Logger._listeners = {} -- optional callbacks, e.g. to feed the UI

local LEVELS = {
    INFO = "INFO",
    WARN = "WARN",
    ERROR = "ERROR",
    SUCCESS = "SUCCESS",
    DEBUG = "DEBUG",
}
Logger.Levels = LEVELS

local function timestamp()
    local ok, result = pcall(function()
        return os.date("%H:%M:%S")
    end)
    return ok and result or "??:??:??"
end

local function pushToBuffer(line)
    table.insert(Logger._buffer, line)
    local maxLines = Config.Get("Logger.MaxLines") or 200
    if #Logger._buffer > maxLines then
        table.remove(Logger._buffer, 1)
    end
end

local function notifyListeners(entry)
    for _, listener in ipairs(Logger._listeners) do
        pcall(listener, entry)
    end
end

--- Core log function. Never throws.
function Logger.Log(level, module, message)
    if not Config.Get("Logger.Enabled") then
        return
    end

    level = level or LEVELS.INFO
    module = module or "Unknown"
    message = message or ""

    local line = string.format("[%s] [%s] [%s] %s", timestamp(), level, module, tostring(message))

    pcall(pushToBuffer, line)

    if Config.Get("Logger.PrintToConsole") then
        pcall(print, line)
    end

    pcall(notifyListeners, {
        level = level,
        module = module,
        message = message,
        line = line,
        time = timestamp(),
    })
end

function Logger.Info(module, message)
    Logger.Log(LEVELS.INFO, module, message)
end

function Logger.Warn(module, message)
    Logger.Log(LEVELS.WARN, module, message)
end

function Logger.Error(module, message)
    Logger.Log(LEVELS.ERROR, module, message)
end

function Logger.Success(module, message)
    Logger.Log(LEVELS.SUCCESS, module, message)
end

function Logger.Debug(module, message)
    if Config.Get("General.DebugMode") then
        Logger.Log(LEVELS.DEBUG, module, message)
    end
end

--- Subscribe to new log entries (e.g. UI module hooks in here)
function Logger.Subscribe(callback)
    if type(callback) ~= "function" then
        return false, "Listener must be a function"
    end
    table.insert(Logger._listeners, callback)
    return true
end

--- Remove a previously registered listener by identity. Call this on
--- teardown (e.g. UI.Shutdown) so listeners don't accumulate across
--- repeated init/shutdown cycles.
function Logger.Unsubscribe(callback)
    for i, existing in ipairs(Logger._listeners) do
        if existing == callback then
            table.remove(Logger._listeners, i)
            return true
        end
    end
    return false
end

--- Get a copy of the current log buffer
function Logger.GetBuffer()
    local copy = {}
    for i, line in ipairs(Logger._buffer) do
        copy[i] = line
    end
    return copy
end

function Logger.Clear()
    Logger._buffer = {}
end

--[[
    Lumexa - Utilities Module
    Shared helper functions used across other modules.
    All functions are defensive: bad input returns nil/false + reason
    instead of throwing where reasonably possible.
]]


local Utilities = {}

--- Wraps a function call in pcall and logs on failure.
--- Returns success (bool), result (any)
function Utilities.SafeCall(fn, moduleName, ...)
    if type(fn) ~= "function" then
        Logger.Error(moduleName or "Utilities", "SafeCall received a non-function")
        return false, nil
    end

    local ok, result = pcall(fn, ...)
    if not ok then
        Logger.Error(moduleName or "Utilities", "SafeCall failed: " .. tostring(result))
        return false, nil
    end

    return true, result
end

--- Deep copy a table (no metatables, plain data only)
function Utilities.DeepCopy(original)
    if type(original) ~= "table" then
        return original
    end

    local copy = {}
    for key, value in pairs(original) do
        if type(value) == "table" then
            copy[key] = Utilities.DeepCopy(value)
        else
            copy[key] = value
        end
    end
    return copy
end

--- Merge table b into table a (a is not mutated; returns new table)
function Utilities.MergeTables(a, b)
    local result = Utilities.DeepCopy(a) or {}
    if type(b) == "table" then
        for key, value in pairs(b) do
            result[key] = value
        end
    end
    return result
end

--- Clamp a number between min and max
function Utilities.Clamp(value, min, max)
    if type(value) ~= "number" then
        return min
    end
    if value < min then return min end
    if value > max then return max end
    return value
end

--- Round a number to N decimal places
function Utilities.Round(value, decimals)
    decimals = decimals or 0
    local mult = 10 ^ decimals
    return math.floor(value * mult + 0.5) / mult
end

--- Distance between two Vector3-like tables {x,y,z} without requiring roblox types,
--- but also supports real Vector3 if Magnitude is available.
function Utilities.Distance(a, b)
    if not a or not b then
        return math.huge
    end

    -- Prefer native Vector3 math if present
    if type((a - b)) == "userdata" then
        local ok, mag = pcall(function() return (a - b).Magnitude end)
        if ok then return mag end
    end

    local dx = (a.x or a.X or 0) - (b.x or b.X or 0)
    local dy = (a.y or a.Y or 0) - (b.y or b.Y or 0)
    local dz = (a.z or a.Z or 0) - (b.z or b.Z or 0)
    return math.sqrt(dx * dx + dy * dy + dz * dz)
end

--- Simple debounce helper. Returns a function that only allows
--- execution once per `interval` seconds.
function Utilities.Debounce(fn, interval)
    local lastRun = 0
    return function(...)
        local now = os.clock()
        if now - lastRun >= interval then
            lastRun = now
            return fn(...)
        end
    end
end

--- Generate a short random ID (not cryptographically secure, just for tagging)
function Utilities.ShortId(length)
    length = length or 6
    local chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    local id = {}
    for i = 1, length do
        local idx = math.random(1, #chars)
        id[i] = chars:sub(idx, idx)
    end
    return table.concat(id)
end

--- Trim whitespace from a string
function Utilities.Trim(s)
    if type(s) ~= "string" then return s end
    return s:match("^%s*(.-)%s*$")
end

--- Check if a table is empty
function Utilities.IsEmpty(t)
    if type(t) ~= "table" then return true end
    return next(t) == nil
end

-- Scanner

--[[
    Lumexa - Scanner Module
    Periodically scans the workspace for relevant entities
    (bosses, events, NPCs). Other modules subscribe to its results
    rather than scanning the workspace themselves, so there is
    a single source of truth and one scan loop.

    Part 2 additions:
    - Sea detection (via configurable PlaceId -> Sea name map)
    - Boss/event scanning against the configured target list
    - Server metadata collection (JobId, PlaceId, GameId, Players, Timestamp, Uptime)
    - Cached container reference to avoid full-workspace walks every pass
]]


local Scanner = {}

Scanner._running = false
Scanner._listeners = {}
Scanner._lastResults = {}
Scanner._startTime = nil

local Players = nil
local Workspace = nil
pcall(function()
    Players = game:GetService("Players")
    Workspace = game:GetService("Workspace")
end)

-- Cached scan container. Defaults to Workspace itself, but can be narrowed
-- via Config.Scanner.ContainerPath (dot-path under Workspace) so repeated
-- scans don't walk unrelated parts of the tree every pass.
Scanner._cachedContainer = nil
Scanner._cachedContainerPath = nil

local function resolveContainer(pathString)
    if not Workspace then
        return nil
    end
    if not pathString or pathString == "" then
        return Workspace
    end

    local node = Workspace
    for key in string.gmatch(pathString, "[^%.]+") do
        if not node then
            return nil
        end
        node = node:FindFirstChild(key)
    end
    return node
end

--- Returns the cached scan container, re-resolving it only if the
--- configured path changed or the cached instance was destroyed.
local function getScanContainer()
    local configuredPath = Config.Get("Scanner.ContainerPath")

    local needsRefresh = (Scanner._cachedContainer == nil)
        or (configuredPath ~= Scanner._cachedContainerPath)

    if not needsRefresh then
        local stillValid = pcall(function()
            return Scanner._cachedContainer.Parent ~= nil or Scanner._cachedContainer == Workspace
        end)
        needsRefresh = not stillValid
    end

    if needsRefresh then
        local resolved = resolveContainer(configuredPath)
        if not resolved then
            Logger.Warn("Scanner", "ContainerPath '" .. tostring(configuredPath) .. "' not found, falling back to Workspace")
            resolved = Workspace
        end
        Scanner._cachedContainer = resolved
        Scanner._cachedContainerPath = configuredPath
    end

    return Scanner._cachedContainer
end

--- Subscribe to scan results. Callback receives a results table.
function Scanner.Subscribe(callback)
    if type(callback) ~= "function" then
        return false, "Listener must be a function"
    end
    table.insert(Scanner._listeners, callback)
    return true
end

local function notify(results)
    for _, listener in ipairs(Scanner._listeners) do
        pcall(listener, results)
    end
end

--- Detects which sea/world the local server belongs to, using a
--- user-supplied PlaceId -> name map (Config.Scanner.SeaMap).
--- Returns nil rather than guessing if no mapping is configured —
--- Lumexa does not assume game-specific IDs on its own.
function Scanner.DetectSea()
    local ok, placeId = pcall(function() return game.PlaceId end)
    if not ok then
        return nil
    end

    local seaMap = Config.Get("Scanner.SeaMap") or {}
    local detected = seaMap[placeId] or seaMap[tostring(placeId)]

    if detected then
        return detected
    end

    Logger.Debug("Scanner", "No sea mapping for PlaceId " .. tostring(placeId))
    return nil
end

--- Collects server/session metadata. All fields are pcall-guarded
--- individually so one failure doesn't blank out the rest.
function Scanner.GetServerInfo()
    local info = {
        jobId = nil,
        placeId = nil,
        gameId = nil,
        players = nil,
        timestamp = os.time(),
        uptime = Scanner._startTime and (os.clock() - Scanner._startTime) or 0,
    }

    pcall(function() info.jobId = game.JobId end)
    pcall(function() info.placeId = game.PlaceId end)
    pcall(function() info.gameId = game.GameId end)
    pcall(function() info.players = Players and #Players:GetPlayers() or nil end)

    return info
end

--- Checks whether an entity name matches the configured boss/event target list.
local function isTargetName(name)
    local targets = Config.Get("BossManager.TargetBosses") or {}
    for _, targetName in ipairs(targets) do
        if targetName == name then
            return true
        end
    end
    return false
end

--- Performs a single scan pass. Safe to call directly for a one-off scan.
--- Collects ALL matching bosses/events in the pass before returning, so
--- downstream consumers (BossManager, ServerHop) can decide what to do
--- with the full set rather than reacting to just the first hit.
function Scanner.ScanOnce()
    Status.Set(Status.States.SEARCHING)

    local results = {
        entities = {},
        bosses = {},
        sea = nil,
        serverInfo = nil,
        scannedAt = os.clock(),
    }

    local ok, err = pcall(function()
        local container = getScanContainer()
        if not container then return end

        local maxDistance = Config.Get("Scanner.MaxDistance") or 500
        local localPlayer = Players and Players.LocalPlayer
        local rootPart = localPlayer
            and localPlayer.Character
            and localPlayer.Character:FindFirstChild("HumanoidRootPart")

        for _, obj in ipairs(container:GetDescendants()) do
            if obj:IsA("Model") and obj:FindFirstChild("Humanoid") then
                local primary = obj.PrimaryPart or obj:FindFirstChild("HumanoidRootPart")
                if primary then
                    local distance = rootPart
                        and Utilities.Distance(rootPart.Position, primary.Position)
                        or 0

                    if distance <= maxDistance then
                        local entry = {
                            name = obj.Name,
                            instance = obj,
                            distance = distance,
                        }
                        table.insert(results.entities, entry)

                        if isTargetName(obj.Name) then
                            table.insert(results.bosses, entry)
                        end
                    end
                end
            end
        end
    end)

    if not ok then
        Logger.Error("Scanner", "ScanOnce failed: " .. tostring(err))
        return Scanner._lastResults
    end

    -- Sea + server metadata are cheap, so gather them every pass rather
    -- than caching stale values.
    results.sea = Scanner.DetectSea()
    results.serverInfo = Scanner.GetServerInfo()

    if #results.bosses > 1 then
        Logger.Debug("Scanner", string.format("%d target bosses detected this pass", #results.bosses))
    end

    Status.Set(#results.bosses > 0 and Status.States.BOSS_FOUND or Status.States.NO_BOSS_FOUND)

    Scanner._lastResults = results
    notify(results)
    return results
end

--- Starts the periodic scan loop on its own coroutine/thread.
function Scanner.Start()
    if Scanner._running then
        Logger.Warn("Scanner", "Start called but Scanner is already running")
        return false
    end

    Scanner._running = true
    Scanner._startTime = os.clock()
    Logger.Info("Scanner", "Scanner started")

    task.spawn(function()
        while Scanner._running do
            Utilities.SafeCall(Scanner.ScanOnce, "Scanner")
            local interval = Config.Get("Scanner.Interval") or 1
            task.wait(interval)
        end
    end)

    return true
end

function Scanner.Stop()
    if not Scanner._running then
        return false
    end
    Scanner._running = false
    Logger.Info("Scanner", "Scanner stopped")
    return true
end

function Scanner.GetLastResults()
    return Scanner._lastResults
end


-- ============================================================
-- Webhook
-- ============================================================
--[[
    Lumexa - Webhook Module
    Sends messages/embeds to user-supplied Discord webhook URLs.
    URLs are NEVER hardcoded here — they must be set via
    Config.Set("Webhook.URL", "...") or Config.Webhooks[key] = "..."
    at runtime by the user.

    Part 3 additions:
    - Config.Webhooks named mapping (route different events to different channels)
    - Embed generation helper
    - Queue with sequential (one-at-a-time) processing
    - Retry system with configurable attempts/delay
    - Duplicate protection keyed on JobId + boss name
]]


local Webhook = {}

Webhook._lastSent = 0
Webhook._totalSent = 0
Webhook._queue = {}
Webhook._sentDedupeKeys = {} -- dedupeKey -> timestamp (os.clock()) of when it was sent
Webhook._processing = false
Webhook._running = false

local HttpService = nil
pcall(function()
    HttpService = game:GetService("HttpService")
end)

--- Validates that a webhook URL looks like a real Discord webhook.
--- This is a shape check only, not a guarantee of validity.
local function isValidWebhookUrl(url)
    if type(url) ~= "string" then return false end
    return url:match("^https://discord%.com/api/webhooks/") ~= nil
        or url:match("^https://discordapp%.com/api/webhooks/") ~= nil
end

--- Resolves a URL for a given webhook key. Falls back to the legacy
--- single Config.Webhook.URL if no keyed entry exists.
local function resolveUrl(webhookKey)
    local webhooks = Config.Get("Webhooks") or {}
    local url = webhookKey and webhooks[webhookKey]

    if not isValidWebhookUrl(url) then
        url = Config.Get("Webhook.URL")
    end

    return url
end

--- Internal rate limiter based on Config.Webhook.RateLimitSeconds
local function canSendNow()
    local rateLimit = Config.Get("Webhook.RateLimitSeconds") or 3
    return (os.clock() - Webhook._lastSent) >= rateLimit
end

--- Build a structured embed table. Pure function, does no sending.
function Webhook.BuildEmbed(title, description, fields, color)
    local embedFields = {}
    for _, field in ipairs(fields or {}) do
        table.insert(embedFields, {
            name = tostring(field.name or "Field"),
            value = tostring(field.value or ""),
            inline = field.inline ~= false,
        })
    end

    local isoTime
    local ok, result = pcall(function() return DateTime.now():ToIsoDate() end)
    isoTime = ok and result or os.date("!%Y-%m-%dT%H:%M:%SZ")

    return {
        title = title or "Lumexa",
        description = description or "",
        color = color or 0x5865F2,
        fields = embedFields,
        timestamp = isoTime,
    }
end

--- Convenience: build a boss-sighting embed with a JobId+Boss dedupe key baked in.
function Webhook.BuildBossEmbed(bossName, extraFields)
    local jobId = "unknown"
    pcall(function() jobId = game.JobId end)

    local fields = { { name = "Server", value = jobId, inline = true } }
    for _, f in ipairs(extraFields or {}) do
        table.insert(fields, f)
    end

    local embed = Webhook.BuildEmbed("Boss Detected: " .. tostring(bossName), nil, fields, 0xE74C3C)
    local dedupeKey = jobId .. "_" .. tostring(bossName)
    return embed, dedupeKey
end

local function rawPost(url, payload)
    local ok, encoded = pcall(function()
        return HttpService:JSONEncode(payload)
    end)
    if not ok then
        return false, "Encoding failed: " .. tostring(encoded)
    end

    -- Roblox's PostAsync has no built-in configurable timeout, so we
    -- race it against a deadline: fire the request on its own thread
    -- and stop waiting on it once TimeoutSeconds elapses. The request
    -- may still complete in the background after we give up, but the
    -- caller is freed to retry/move on rather than hang indefinitely.
    local timeoutSeconds = Config.Get("Webhook.TimeoutSeconds") or 10
    local done = false
    local success, requestErr

    task.spawn(function()
        success, requestErr = pcall(function()
            HttpService:PostAsync(url, encoded, Enum.HttpContentType.ApplicationJson)
        end)
        done = true
    end)

    local waited = 0
    local pollInterval = 0.1
    while not done and waited < timeoutSeconds do
        task.wait(pollInterval)
        waited = waited + pollInterval
    end

    if not done then
        Logger.Warn("Webhook", string.format("Request exceeded %ds timeout, treating as failed", timeoutSeconds))
        return false, "Timed out after " .. timeoutSeconds .. "s"
    end

    if not success then
        return false, "Request failed: " .. tostring(requestErr)
    end

    return true, nil
end

--- Enqueue an embed for sending. Skips silently (returns false, reason)
--- if a dedupeKey was already sent successfully or is already queued.
--- @param webhookKey string|nil which Config.Webhooks[] entry to use
--- @param embed table  an embed built via Webhook.BuildEmbed
--- @param dedupeKey string|nil unique key, e.g. JobId .. "_" .. bossName
function Webhook.Enqueue(webhookKey, embed, dedupeKey)
    if not Config.Get("Webhook.Enabled") then
        return false, "Webhook is disabled in config"
    end

    if dedupeKey then
        if Webhook._sentDedupeKeys[dedupeKey] then
            Logger.Debug("Webhook", "Skipped duplicate (already sent): " .. dedupeKey)
            return false, "Duplicate: already sent"
        end
        for _, item in ipairs(Webhook._queue) do
            if item.dedupeKey == dedupeKey then
                Logger.Debug("Webhook", "Skipped duplicate (already queued): " .. dedupeKey)
                return false, "Duplicate: already queued"
            end
        end
    end

    table.insert(Webhook._queue, {
        webhookKey = webhookKey,
        embed = embed,
        dedupeKey = dedupeKey,
        attempts = 0,
    })

    Logger.Debug("Webhook", "Queued embed" .. (dedupeKey and (" [" .. dedupeKey .. "]") or ""))
    return true, nil
end

--- Convenience wrapper for boss sightings specifically. If webhookKey
--- isn't given, defaults to the boss name — so adding a new boss is
--- as simple as BossManager.AddTarget(name) plus, optionally, a
--- Config.Webhooks[name] entry to route it to its own channel.
function Webhook.EnqueueBossAlert(bossName, extraFields, webhookKey)
    local embed, dedupeKey = Webhook.BuildBossEmbed(bossName, extraFields)
    return Webhook.Enqueue(webhookKey or bossName, embed, dedupeKey)
end

--- Attempts to send one queue item, respecting retry config.
--- Returns true if the item is done (sent or permanently failed/dropped).
local function processItem(item)
    local url = resolveUrl(item.webhookKey)
    if not isValidWebhookUrl(url) then
        Logger.Warn("Webhook", "Dropping queued item: no valid webhook URL for key " .. tostring(item.webhookKey))
        return true -- drop, nothing more we can do
    end

    if not HttpService then
        Logger.Error("Webhook", "HttpService unavailable, dropping queue item")
        return true
    end

    if not canSendNow() then
        return false -- not done, retry on a later tick once rate limit clears
    end

    Status.Set(Status.States.SENDING_WEBHOOK)
    local ok, err = rawPost(url, { embeds = { item.embed } })
    Webhook._lastSent = os.clock()

    if ok then
        if item.dedupeKey then
            Webhook._sentDedupeKeys[item.dedupeKey] = os.clock()
        end
        Webhook._totalSent = Webhook._totalSent + 1
        Status.Set(Status.States.WEBHOOK_SENT)
        Logger.Success("Webhook", "Embed sent successfully" .. (item.dedupeKey and (" [" .. item.dedupeKey .. "]") or ""))
        return true
    end

    item.attempts = item.attempts + 1
    local maxRetries = Config.Get("Webhook.MaxRetries") or 3

    if item.attempts >= maxRetries then
        Logger.Error("Webhook", string.format("Giving up after %d attempts: %s", item.attempts, tostring(err)))
        return true -- permanently failed, drop it
    end

    Logger.Warn("Webhook", string.format("Send failed (attempt %d/%d): %s", item.attempts, maxRetries, tostring(err)))
    Status.Set(Status.States.RETRYING)
    return false -- keep in queue, will be retried
end

--- Purges dedupe entries older than Config.Cache.LifetimeSeconds.
--- Piggybacks on the queue processor's own loop rather than spawning
--- a second thread, per Part 8's "only one master loop" guidance.
local function expireOldEntries()
    local lifetime = Config.Get("Cache.LifetimeSeconds") or 3600
    local now = os.clock()
    local expired = {}

    for key, sentAt in pairs(Webhook._sentDedupeKeys) do
        if (now - sentAt) > lifetime then
            table.insert(expired, key)
        end
    end

    for _, key in ipairs(expired) do
        Webhook._sentDedupeKeys[key] = nil
    end

    if #expired > 0 then
        Logger.Debug("Webhook", string.format("Expired %d old dedupe entr%s", #expired, #expired == 1 and "y" or "ies"))
    end
end

--- Starts the sequential queue processor. Only one item is ever
--- in flight at a time — no concurrent sends.
function Webhook.StartQueueProcessor()
    if Webhook._running then
        return false, "Queue processor already running"
    end

    Webhook._running = true
    Logger.Info("Webhook", "Queue processor started")
    local lastSweep = os.clock()

    task.spawn(function()
        while Webhook._running do
            local sweepInterval = Config.Get("Cache.SweepIntervalSeconds") or 300
            if (os.clock() - lastSweep) >= sweepInterval then
                pcall(expireOldEntries)
                lastSweep = os.clock()
            end

            if #Webhook._queue > 0 and not Webhook._processing then
                Webhook._processing = true

                local item = Webhook._queue[1]
                local retryDelay = Config.Get("Webhook.RetryDelaySeconds") or 5

                local ok, done = pcall(processItem, item)
                if not ok then
                    Logger.Error("Webhook", "processItem crashed: " .. tostring(done))
                    table.remove(Webhook._queue, 1)
                elseif done then
                    table.remove(Webhook._queue, 1)
                else
                    -- Leave it at the front of the queue; wait before
                    -- the next tick retries it (covers both rate-limit
                    -- backoff and failed-send backoff).
                    task.wait(retryDelay)
                end

                Webhook._processing = false
            end

            local interval = Config.Get("Webhook.QueueIntervalSeconds") or 1
            task.wait(interval)
        end
    end)

    return true
end

function Webhook.StopQueueProcessor()
    Webhook._running = false
    Logger.Info("Webhook", "Queue processor stopped")
end

function Webhook.GetQueueLength()
    return #Webhook._queue
end

--- Synchronously drains the queue, bypassing the normal tick interval.
--- Intended for graceful shutdown (game:BindToClose): attempts to get
--- every pending webhook out before the server closes, bounded by
--- maxSeconds so it can't hang the shutdown indefinitely.
function Webhook.FlushQueue(maxSeconds)
    maxSeconds = maxSeconds or 8
    local deadline = os.clock() + maxSeconds

    Logger.Info("Webhook", string.format("Flushing %d queued webhook(s) before shutdown", #Webhook._queue))

    while #Webhook._queue > 0 and os.clock() < deadline do
        local item = Webhook._queue[1]
        local ok, done = pcall(processItem, item)

        if not ok then
            Logger.Error("Webhook", "processItem crashed during flush: " .. tostring(done))
            table.remove(Webhook._queue, 1)
        elseif done then
            table.remove(Webhook._queue, 1)
        else
            task.wait(0.2)
        end
    end

    if #Webhook._queue > 0 then
        Logger.Warn("Webhook", string.format("Shutdown flush timed out with %d item(s) still queued", #Webhook._queue))
    else
        Logger.Info("Webhook", "Queue fully flushed before shutdown")
    end
end

--- Send a plain text message immediately (bypasses the queue).
--- Kept for simple one-off notifications outside the boss/event flow.
function Webhook.Send(message, webhookKey)
    if not Config.Get("Webhook.Enabled") then
        return false, "Webhook is disabled in config"
    end

    local url = resolveUrl(webhookKey)
    if not isValidWebhookUrl(url) then
        Logger.Warn("Webhook", "Send blocked: no valid webhook URL configured")
        return false, "No valid webhook URL configured"
    end

    if not canSendNow() then
        Logger.Debug("Webhook", "Send skipped due to rate limit")
        return false, "Rate limited"
    end

    if not HttpService then
        Logger.Error("Webhook", "HttpService unavailable")
        return false, "HttpService unavailable"
    end

    local ok, err = rawPost(url, { content = tostring(message) })
    if ok then
        Webhook._lastSent = os.clock()
        Webhook._totalSent = Webhook._totalSent + 1
        Logger.Info("Webhook", "Message sent successfully")
        return true, nil
    end

    Logger.Error("Webhook", tostring(err))
    return false, err
end

--- Send a structured embed immediately (bypasses the queue).
--- Prefer Webhook.Enqueue / EnqueueBossAlert for anything that should
--- get retry + duplicate protection + sequential ordering.
function Webhook.SendEmbed(title, description, fields, webhookKey)
    if not Config.Get("Webhook.Enabled") then
        return false, "Webhook is disabled in config"
    end

    local url = resolveUrl(webhookKey)
    if not isValidWebhookUrl(url) then
        Logger.Warn("Webhook", "SendEmbed blocked: no valid webhook URL configured")
        return false, "No valid webhook URL configured"
    end

    if not canSendNow() then
        return false, "Rate limited"
    end

    if not HttpService then
        return false, "HttpService unavailable"
    end

    local embed = Webhook.BuildEmbed(title, description, fields)
    local ok, err = rawPost(url, { embeds = { embed } })
    if ok then
        Webhook._lastSent = os.clock()
        Webhook._totalSent = Webhook._totalSent + 1
        Logger.Info("Webhook", "Embed sent successfully")
        return true, nil
    end

    Logger.Error("Webhook", tostring(err))
    return false, err
end

--- Cumulative count of successfully sent webhooks (queue + immediate combined)
function Webhook.GetTotalSent()
    return Webhook._totalSent
end

-- SeaManager
-- ============================================================
--[[
    Lumexa - SeaManager Module
    Tracks which "sea" the player is currently in and exposes
    helpers for sea-dependent logic. Does not assume any
    default sea — the user must configure PreferredSea.
]]


local SeaManager = {}

SeaManager._currentSea = nil
SeaManager._listeners = {}

local Players = nil
pcall(function()
    Players = game:GetService("Players")
end)

--- Subscribe to sea-change events. Callback receives (newSea, oldSea).
function SeaManager.Subscribe(callback)
    if type(callback) ~= "function" then
        return false, "Listener must be a function"
    end
    table.insert(SeaManager._listeners, callback)
    return true
end

local function notify(newSea, oldSea)
    for _, listener in ipairs(SeaManager._listeners) do
        pcall(listener, newSea, oldSea)
    end
end

--- Attempts to detect the current sea from game state.
--- This relies on a PlaceId lookup table since Roblox doesn't
--- expose "sea number" directly. Uses the same Config.Scanner.SeaMap
--- that Scanner.DetectSea() reads, so there is one map to populate,
--- not two.
function SeaManager.DetectCurrentSea()
    local ok, placeId = pcall(function()
        return game.PlaceId
    end)

    if not ok then
        Logger.Error("SeaManager", "Failed to read PlaceId")
        return nil
    end

    local seaMap = Config.Get("Scanner.SeaMap") or {}
    local detected = seaMap[placeId] or seaMap[tostring(placeId)]
    if not detected then
        Logger.Warn("SeaManager", "Unknown PlaceId " .. tostring(placeId) .. " — sea not mapped")
        return nil
    end

    SeaManager.SetCurrentSea(detected)
    return detected
end

--- Manually set the current sea (e.g. after detection or user override)
function SeaManager.SetCurrentSea(seaName)
    if type(seaName) ~= "string" or seaName == "" then
        return false, "Invalid sea name"
    end

    local old = SeaManager._currentSea
    if old == seaName then
        return true -- no change
    end

    SeaManager._currentSea = seaName
    Logger.Info("SeaManager", "Sea changed: " .. tostring(old) .. " -> " .. seaName)
    notify(seaName, old)
    return true
end

function SeaManager.GetCurrentSea()
    return SeaManager._currentSea
end

--- Checks if the current sea matches the user's configured preference
function SeaManager.IsPreferredSea()
    local preferred = Config.Get("SeaManager.PreferredSea")
    if not preferred then
        return false, "No preferred sea configured"
    end
    return SeaManager._currentSea == preferred
end


-- ============================================================
-- BossManager
-- ============================================================
--[[
    Lumexa - BossManager Module
    Tracks configured target bosses using Scanner results.
    Does not scan the workspace itself — stays modular by
    subscribing to Scanner instead of duplicating that work.
]]


local BossManager = {}

BossManager._activeTargets = {}
BossManager._listeners = {}
BossManager._allClearListeners = {}
BossManager._hadTargets = false
BossManager._totalFound = 0

--- Subscribe to boss sighting updates
function BossManager.Subscribe(callback)
    if type(callback) ~= "function" then
        return false, "Listener must be a function"
    end
    table.insert(BossManager._listeners, callback)
    return true
end

--- Subscribe to the moment ALL previously-sighted target bosses are gone.
--- ServerHop (or Main) should wait for this before hopping, so a hop
--- doesn't fire off the back of just the first boss in a multi-boss pass.
function BossManager.SubscribeAllClear(callback)
    if type(callback) ~= "function" then
        return false, "Listener must be a function"
    end
    table.insert(BossManager._allClearListeners, callback)
    return true
end

local function notify(sightings)
    for _, listener in ipairs(BossManager._listeners) do
        pcall(listener, sightings)
    end
end

local function notifyAllClear()
    for _, listener in ipairs(BossManager._allClearListeners) do
        pcall(listener)
    end
end

--- Checks whether a given entity name matches the configured target list.
local function isTargetBoss(name)
    local targets = Config.Get("BossManager.TargetBosses") or {}
    for _, targetName in ipairs(targets) do
        if targetName == name then
            return true
        end
    end
    return false
end

--- Called on every Scanner result. Filters for configured bosses only.
local function onScanResults(results)
    if not Config.Get("BossManager.Enabled") then
        return
    end

    local sightings = {}
    for _, entity in ipairs(results.entities or {}) do
        if isTargetBoss(entity.name) then
            table.insert(sightings, entity)
        end
    end

    BossManager._activeTargets = sightings

    if #sightings > 0 then
        BossManager._hadTargets = true
        Logger.Debug("BossManager", string.format("%d target boss(es) in range", #sightings))
        notify(sightings)

        for _, sighting in ipairs(sightings) do
            local safeOk, enqueuedOk = Utilities.SafeCall(Webhook.EnqueueBossAlert, "BossManager", sighting.name, {
                { name = "Distance", value = tostring(Utilities.Round(sighting.distance, 1)), inline = true },
            })
            if safeOk and enqueuedOk then
                BossManager._totalFound = BossManager._totalFound + 1
            end
        end
    elseif BossManager._hadTargets then
        -- Transition from "had targets" to "none left" — this is the
        -- signal that every boss from the last pass has been handled.
        BossManager._hadTargets = false
        Logger.Info("BossManager", "All target bosses cleared")
        notifyAllClear()
    end
end

--- True if there are no active target bosses right now. Use this (or
--- SubscribeAllClear) to gate ServerHop so it waits for every boss in
--- a multi-boss pass before considering a hop.
function BossManager.AllTargetsCleared()
    return Utilities.IsEmpty(BossManager._activeTargets)
end

Scanner.Subscribe(onScanResults)

--- Add a boss name to the target list
function BossManager.AddTarget(name)
    if type(name) ~= "string" then
        return false, "Invalid boss name"
    end
    name = name:match("^%s*(.-)%s*$") -- trim whitespace from UI text input
    if name == "" then
        return false, "Invalid boss name"
    end

    local targets = Config.Get("BossManager.TargetBosses") or {}
    for _, existing in ipairs(targets) do
        if existing == name then
            return false, "Already tracking " .. name
        end
    end

    table.insert(targets, name)
    Config.Set("BossManager.TargetBosses", targets)
    Logger.Info("BossManager", "Now tracking: " .. name)
    return true
end

--- Remove a boss name from the target list
function BossManager.RemoveTarget(name)
    if type(name) == "string" then
        name = name:match("^%s*(.-)%s*$")
    end
    local targets = Config.Get("BossManager.TargetBosses") or {}
    for i, existing in ipairs(targets) do
        if existing == name then
            table.remove(targets, i)
            Config.Set("BossManager.TargetBosses", targets)
            Logger.Info("BossManager", "Stopped tracking: " .. name)
            return true
        end
    end
    return false, "Not currently tracking " .. name
end

--- Get the closest currently active target, if any
function BossManager.GetClosestTarget()
    if Utilities.IsEmpty(BossManager._activeTargets) then
        return nil
    end

    local closest = nil
    for _, target in ipairs(BossManager._activeTargets) do
        if not closest or target.distance < closest.distance then
            closest = target
        end
    end
    return closest
end

function BossManager.GetActiveTargets()
    return BossManager._activeTargets
end

--- Cumulative count of unique boss detections (JobId+name deduped via Webhook)
function BossManager.GetTotalFound()
    return BossManager._totalFound
end


-- ============================================================
-- ServerHop
-- ============================================================
--[[
    Lumexa - ServerHop Module
    Handles hopping to a new server instance based on configured
    player-count thresholds. Uses Roblox's official public server
    listing API rather than any undocumented endpoint.

    Part 4 additions:
    - Paginated ("infinite" cursor-following) server scanning
    - Skips any JobId already visited this session, not just the current one
    - Prefers the candidate with the most available slots
    - Retries the teleport call itself, not just the listing fetch
]]


local ServerHop = {}

ServerHop._hopping = false
ServerHop._visitedJobIds = {} -- JobId -> timestamp (os.clock()) of when it was visited/tried

local TeleportService = nil
local HttpService = nil
local Players = nil

pcall(function()
    TeleportService = game:GetService("TeleportService")
    HttpService = game:GetService("HttpService")
    Players = game:GetService("Players")
end)

-- Mark our own starting server as visited so we never "hop" to ourselves.
pcall(function()
    ServerHop._visitedJobIds[game.JobId] = os.clock()
end)

--- Fetches a single page of public server instances for the current place.
--- Returns (servers, nextCursor) or (nil, nil) on failure.
local function fetchServerPage(cursor)
    if not HttpService then
        Logger.Error("ServerHop", "HttpService unavailable")
        return nil, nil
    end

    local placeId = game.PlaceId
    local url = string.format(
        "https://games.roblox.com/v1/games/%d/servers/Public?sortOrder=Asc&limit=100",
        placeId
    )
    if cursor and cursor ~= "" then
        url = url .. "&cursor=" .. HttpService:UrlEncode(cursor)
    end

    local ok, response = pcall(function()
        return HttpService:GetAsync(url)
    end)

    if not ok then
        Logger.Error("ServerHop", "Failed to fetch server list: " .. tostring(response))
        return nil, nil
    end

    local decodeOk, data = pcall(function()
        return HttpService:JSONDecode(response)
    end)

    if not decodeOk or not data or not data.data then
        Logger.Error("ServerHop", "Failed to decode server list")
        return nil, nil
    end

    return data.data, data.nextPageCursor
end

--- Returns true if a server is a valid hop candidate: not visited,
--- has room, and falls within the configured player-count thresholds.
local function isCandidate(server)
    local minPlayers = Config.Get("ServerHop.MinPlayers") or 0
    local maxPlayers = Config.Get("ServerHop.MaxPlayers") or 100

    if ServerHop._visitedJobIds[server.id] then
        return false
    end
    if server.playing >= server.maxPlayers then
        return false -- full
    end
    if server.playing < minPlayers or server.playing > maxPlayers then
        return false
    end
    return true
end

--- Scans pages of the public server list (following the cursor) until
--- either a good candidate is found or the list/safety cap is exhausted.
--- Among everything seen, prefers the server with the most available slots.
local function findBestServer()
    local cursor = nil
    local bestServer = nil
    local bestAvailable = -1
    local maxPages = Config.Get("ServerHop.MaxPagesToScan") or 25
    local pagesScanned = 0

    repeat
        local servers, nextCursor = fetchServerPage(cursor)
        pagesScanned = pagesScanned + 1

        if not servers then
            break -- fetch failed, stop scanning this attempt
        end

        for _, server in ipairs(servers) do
            if isCandidate(server) then
                local available = server.maxPlayers - server.playing
                if available > bestAvailable then
                    bestAvailable = available
                    bestServer = server
                end
            end
        end

        cursor = nextCursor
    until not cursor or cursor == "" or pagesScanned >= maxPages

    if cursor and cursor ~= "" and pagesScanned >= maxPages then
        Logger.Warn("ServerHop", string.format("Stopped scanning after safety cap of %d pages", maxPages))
    end

    return bestServer, pagesScanned
end

--- Attempts the actual teleport call, retrying on failure per config.
local function teleportWithRetry(target)
    local maxRetries = Config.Get("ServerHop.MaxTeleportRetries") or 3
    local retryDelay = Config.Get("ServerHop.TeleportRetryDelaySeconds") or 3

    for attempt = 1, maxRetries do
        local ok, err = pcall(function()
            TeleportService:TeleportToPlaceInstance(game.PlaceId, target.id, Players.LocalPlayer)
        end)

        if ok then
            return true, nil
        end

        Logger.Warn("ServerHop", string.format("Teleport attempt %d/%d failed: %s", attempt, maxRetries, tostring(err)))
        Status.Set(Status.States.TELEPORT_FAILED)

        if attempt < maxRetries then
            Status.Set(Status.States.RETRYING)
            task.wait(retryDelay)
        end
    end

    return false, "Teleport failed after " .. tostring(maxRetries) .. " attempts"
end

--- Attempts a single server hop. Returns success (bool), reason (string)
function ServerHop.HopOnce()
    if not Config.Get("ServerHop.Enabled") then
        return false, "ServerHop is disabled in config"
    end

    if ServerHop._hopping then
        return false, "Already hopping"
    end

    if not TeleportService then
        return false, "TeleportService unavailable"
    end

    ServerHop._hopping = true
    Status.Set(Status.States.SERVER_HOPPING)

    -- Everything risky runs inside this pcall so _hopping is *guaranteed*
    -- to reset below no matter what happens — an uncaught error here used
    -- to leave the flag stuck true forever, permanently blocking all
    -- future hops with "Already hopping".
    local crashed, ok, err = pcall(function()
        local target, pagesScanned = findBestServer()
        if not target then
            Logger.Warn("ServerHop", string.format("No suitable server found after scanning %d page(s)", pagesScanned or 0))
            return false, "No suitable server found"
        end

        Logger.Info("ServerHop", string.format(
            "Selected server %s (%d/%d players, %d slot(s) free) after scanning %d page(s)",
            tostring(target.id), target.playing, target.maxPlayers, target.maxPlayers - target.playing, pagesScanned or 0
        ))

        local teleportOk, teleportErr = teleportWithRetry(target)
        if teleportOk then
            ServerHop.MarkVisited(target.id)
            Logger.Info("ServerHop", "Hopping to server " .. tostring(target.id))
        end
        return teleportOk, teleportErr
    end)

    ServerHop._hopping = false

    if not crashed then
        Logger.Error("ServerHop", "HopOnce crashed: " .. tostring(ok))
        return false, "Internal error: " .. tostring(ok)
    end

    if not ok then
        Logger.Error("ServerHop", tostring(err))
        return false, err
    end

    return true, nil
end

--- Convenience wrapper with the configured delay applied before hopping.
function ServerHop.HopWithDelay()
    local delay = Config.Get("ServerHop.HopDelay") or 5
    Status.Set(Status.States.WAITING)
    task.spawn(function()
        task.wait(delay)
        Utilities.SafeCall(ServerHop.HopOnce, "ServerHop")
    end)
end

--- Manually mark a JobId as visited (e.g. if learned via Scanner.GetServerInfo
--- from a different source) without going through HopOnce.
function ServerHop.MarkVisited(jobId)
    if not jobId then return end

    -- Defensive cap: in a very long single-script-lifetime session that
    -- never actually teleports away (e.g. hopping stays disabled while
    -- scanning runs for hours), this set would otherwise grow forever.
    -- Wipe and start over past a generous threshold rather than leak.
    local maxVisited = 5000
    if ServerHop.GetVisitedCount() >= maxVisited then
        Logger.Warn("ServerHop", "Visited-JobId cache exceeded " .. maxVisited .. ", clearing")
        ServerHop._visitedJobIds = {}
        pcall(function() ServerHop._visitedJobIds[game.JobId] = os.clock() end)
    end

    ServerHop._visitedJobIds[jobId] = os.clock()
end

function ServerHop.GetVisitedCount()
    local count = 0
    for _ in pairs(ServerHop._visitedJobIds) do
        count = count + 1
    end
    return count
end

--- Purges visited-JobId entries older than Config.Cache.LifetimeSeconds.
--- Piggybacks on Scanner's own scan loop (via subscription below)
--- rather than spawning a dedicated thread, keeping to a single
--- master loop per Part 8's thread-management guidance.
local function expireVisited()
    local lifetime = Config.Get("Cache.LifetimeSeconds") or 3600
    local now = os.clock()
    local expired = {}

    for jobId, visitedAt in pairs(ServerHop._visitedJobIds) do
        if (now - visitedAt) > lifetime then
            table.insert(expired, jobId)
        end
    end

    for _, jobId in ipairs(expired) do
        ServerHop._visitedJobIds[jobId] = nil
    end

    if #expired > 0 then
        Logger.Debug("ServerHop", string.format("Expired %d old visited-JobId entr%s", #expired, #expired == 1 and "y" or "ies"))
    end
end

do
    local lastSweep = os.clock()
    Scanner.Subscribe(function()
        local sweepInterval = Config.Get("Cache.SweepIntervalSeconds") or 300
        if (os.clock() - lastSweep) >= sweepInterval then
            pcall(expireVisited)
            lastSweep = os.clock()
        end
    end)
end

--[[
    Lumexa - UI Module
    Builds the interface using Rayfield. Kept separate from all
    logic modules — UI only reads/writes Config and calls public
    functions on other modules, it never contains business logic.

    standards.

local UI = {}

UI._window = nil
UI._tabs = {}
UI._initialized = false
UI._dashboardRunning = false
UI._logListener = nil

local function safeRequireRayfield()
    local ok, Rayfield = pcall(function()
        return loadstring(game:HttpGet('https://sirius.menu/rayfield'))()
    end)

    if not ok then
        Logger.Error("UI", "Failed to load Rayfield: " .. tostring(Rayfield))
        return nil
    end

    return Rayfield
end

function UI.Init()
    if UI._initialized then
        Logger.Warn("UI", "UI.Init called again but UI is already built — skipping to avoid duplicate windows/listeners")
        return true
    end

    local Rayfield = safeRequireRayfield()
    if not Rayfield then
        return false, "Rayfield failed to load"
    end

    local ok, err = pcall(function()
        UI._window = Rayfield:CreateWindow({
            Name = "Lumexa",
            LoadingTitle = "Lumexa",
            LoadingSubtitle = "Modular Framework",
            -- Rayfield's "Default" theme is a dark, glass-panel look with
            -- rounded corners (UICorner) and a violet/purple accent out
            -- of the box, matching the Part 5 premium UI spec without
            -- needing a custom hex theme table.
            Theme = "Default",
            ConfigurationSaving = {
                Enabled = true,
                FolderName = "Lumexa",
                FileName = "Config",
            },
        })

        UI._tabs.Dashboard = UI._window:CreateTab("Dashboard")
        UI._tabs.Main = UI._window:CreateTab("Main")
        UI._tabs.Boss = UI._window:CreateTab("Boss")
        UI._tabs.Sea = UI._window:CreateTab("Sea")
        UI._tabs.ServerHop = UI._window:CreateTab("Server Hop")
        UI._tabs.Webhook = UI._window:CreateTab("Webhook")
        UI._tabs.Logs = UI._window:CreateTab("Logs")

        UI._buildDashboardTab()
        UI._buildMainTab()
        UI._buildBossTab()
        UI._buildSeaTab()
        UI._buildServerHopTab()
        UI._buildWebhookTab()
        UI._buildLogsTab()
    end)

    if not ok then
        Logger.Error("UI", "UI construction failed: " .. tostring(err))
        return false, tostring(err)
    end

    Logger.Info("UI", "Interface initialized")
    UI._initialized = true
    return true
end

function UI._buildDashboardTab()
    local tab = UI._tabs.Dashboard

    local fields = {
        Status = tab:CreateLabel("Status: Stopped"),
        Activity = tab:CreateLabel("Activity: Idle"),
        Runtime = tab:CreateLabel("Runtime: 00:00:00"),
        Sea = tab:CreateLabel("Sea: Unknown"),
        Players = tab:CreateLabel("Players: --"),
        JobId = tab:CreateLabel("JobId: --"),
        BossesFound = tab:CreateLabel("Bosses Found: 0"),
        WebhooksSent = tab:CreateLabel("Webhooks Sent: 0"),
    }

    local function refresh()
        local serverInfo = Scanner.GetLastResults() and Scanner.GetLastResults().serverInfo
        serverInfo = serverInfo or Scanner.GetServerInfo()

        pcall(function() fields.Status:Set("Status: " .. Config.GetStatus()) end)
        pcall(function() fields.Activity:Set("Activity: " .. tostring(Status.Get())) end)
        pcall(function()
            local h = math.floor(Config.GetUptime() / 3600)
            local m = math.floor((Config.GetUptime() % 3600) / 60)
            local s = math.floor(Config.GetUptime() % 60)
            fields.Runtime:Set(string.format("Runtime: %02d:%02d:%02d", h, m, s))
        end)
        pcall(function() fields.Sea:Set("Sea: " .. tostring(SeaManager.GetCurrentSea() or "Unknown")) end)
        pcall(function() fields.Players:Set("Players: " .. tostring(serverInfo and serverInfo.players or "--")) end)
        pcall(function() fields.JobId:Set("JobId: " .. tostring(serverInfo and serverInfo.jobId or "--")) end)
        pcall(function() fields.BossesFound:Set("Bosses Found: " .. tostring(BossManager.GetTotalFound())) end)
        pcall(function() fields.WebhooksSent:Set("Webhooks Sent: " .. tostring(Webhook.GetTotalSent())) end)
    end

    refresh()

    UI._dashboardRunning = true
    task.spawn(function()
        while UI._dashboardRunning do
            pcall(refresh)
            task.wait(1)
        end
    end)
end

--- Full teardown: stops the dashboard refresh loop, unsubscribes from
--- Logger, and destroys the Rayfield window if the library supports it.
--- Not called automatically by Main.Stop() — the dashboard is meant to
--- keep reflecting Config.GetStatus() even while the framework itself
--- is stopped. This exists for genuinely unloading Lumexa (e.g. a
--- "Fully Unload" action) without leaving background threads or
--- listeners behind.
function UI.Shutdown()
    UI._dashboardRunning = false

    if UI._logListener then
        Logger.Unsubscribe(UI._logListener)
        UI._logListener = nil
    end

    if UI._window and UI._window.Destroy then
        pcall(function() UI._window:Destroy() end)
    end

    UI._window = nil
    UI._tabs = {}
    UI._initialized = false
    Logger.Info("UI", "UI shut down")
end

-- Main
-- ============================================================
--[[
    Lumexa - Main Module
    Entry point. Wires modules together and starts the framework.
    Contains no business logic of its own — that stays in each
    module for maintainability and future expansion.

    Modular communication (Part 7):
        Every module only knows about the ones below it in this list
        (or Config/Logger/Utilities, which everything can see). Nothing
        reaches back up the chain by requiring a module above it —
        cross-cutting notifications instead flow through pub/sub:

        Config, Logger, Utilities  (shared, no dependencies of their own)
          -> Scanner       (scans, notifies subscribers via Scanner.Subscribe)
          -> BossManager   (subscribes to Scanner; notifies via
                             BossManager.Subscribe / SubscribeAllClear;
                             pushes alerts into Webhook's queue)
          -> SeaManager    (independent; reads Config.Scanner.SeaMap)
          -> Webhook       (queue consumed by anything that calls Enqueue)
          -> ServerHop     (event-driven — Main wires BossManager's
                             all-clear signal to ServerHop.HopWithDelay)
          -> UI            (reads/writes Config; subscribes to Logger;
                             never required by any module above it)

        This is what makes "easy future boss addition" possible: a new
        boss only needs a name added via BossManager.AddTarget (or the
        Boss tab's input field) — Scanner, Webhook, and the Dashboard
        all pick it up automatically with no other module needing to
        change.
]]


local Main = {}

Main._started = false

-- Wired once at module load, not inside Start() — otherwise a
-- Stop() -> Start() cycle would register a duplicate listener every
-- time, same leak class as the UI double-init issue.
-- Only hop once every target boss from a pass has been handled, not
-- after just the first one (see BossManager.SubscribeAllClear).
BossManager.SubscribeAllClear(function()
    if Main._started and Config.Get("ServerHop.Enabled") then
        Logger.Info("Main", "All target bosses cleared, considering server hop")

        task.spawn(function()
            -- Send every queued webhook before hopping, not just queue
            -- them — bounded wait so a stuck webhook can't block hopping
            -- forever (Webhook's own retry/drop logic still applies).
            local deadline = os.clock() + 15
            while Webhook.GetQueueLength() > 0 and os.clock() < deadline do
                task.wait(0.5)
            end

            Utilities.SafeCall(ServerHop.HopWithDelay, "Main")
        end)
    end
end)

UI.SetShutdownHandler(function()
    Main.Shutdown()
end)

--[[
    Startup sequence (Part 7):
        Config  -> already loaded — every module requires it first
        Logger  -> already loaded — active from the moment Config is ready
        UI      -> built so the user has visibility into everything below
        Scanner -> continuous scan loop (sea detection + boss/entity scanning)
        Queue   -> Webhook queue processor, always running so nothing sent
                   by Scanner/BossManager gets dropped on the floor
        Hopper  -> ServerHop itself is event-driven (wired to BossManager's
                   all-clear signal at module load, see below), so its
                   "startup" step here is just announcing readiness
]]
function Main.Start()
    if Main._started then
        Logger.Warn("Main", "Start called but Lumexa is already running")
        return false
    end

    Logger.Info("Main", "Lumexa v" .. Config.Version .. " starting...")

    -- Stage: UI
    local uiOk, uiErr = Utilities.SafeCall(UI.Init, "Main")
    if not uiOk then
        Logger.Error("Main", "UI failed to initialize: " .. tostring(uiErr))
        -- continue anyway — framework can run headless
    end

    -- Stage: Scanner
    Utilities.SafeCall(SeaManager.DetectCurrentSea, "Main")
    if Config.Get("General.AutoStart") then
        Utilities.SafeCall(Scanner.Start, "Main")
    else
        Logger.Info("Main", "AutoStart disabled — Scanner is ready but waiting for manual start")
    end

    -- Stage: Queue
    Utilities.SafeCall(Webhook.StartQueueProcessor, "Main")

    -- Stage: Hopper
    if Config.Get("ServerHop.Enabled") then
        Logger.Info("Main", "ServerHop is enabled and listening for boss-cleared events")
    else
        Logger.Info("Main", "ServerHop is disabled in config")
    end

    Main._started = true
    Config.SetRunning(true)
    Status.Set(Status.States.CONNECTED)
    Logger.Info("Main", "Lumexa started successfully")
    return true
end

--- Seconds Lumexa has been running in total, across start/stop cycles.
--- Freezes while stopped rather than continuing to climb. Delegates
--- to Config so UI (which Main requires) can read status/uptime
--- without needing to require Main back.
function Main.GetUptime()
    return Config.GetUptime()
end

function Main.GetStatus()
    return Config.GetStatus()
end

function Main.Stop()
    if not Main._started then
        return false
    end

    Utilities.SafeCall(Scanner.Stop, "Main")
    Utilities.SafeCall(Webhook.StopQueueProcessor, "Main")

    Main._started = false
    Config.SetRunning(false)
    Logger.Info("Main", "Lumexa stopped")
    return true
end

--- Full teardown: stops everything Main.Stop() does, plus tears down
--- the UI (destroys the window, stops its refresh loop, unsubscribes
--- its Logger listener). Use this to genuinely unload Lumexa rather
--- than just pause it — after this, Main.Start() would build a fresh
--- UI from scratch since UI._initialized is reset.
function Main.Shutdown()
    Main.Stop()
    Utilities.SafeCall(UI.Shutdown, "Main")
    Logger.Info("Main", "Lumexa fully shut down")
    return true
end

-- Combined-build note: the modular version only auto-started if a
-- sibling "_NoAutoStart" marker instance was absent (script.Parent
-- being a folder of ModuleScripts). There's no such hierarchy in a
-- single combined script, so it just starts unconditionally here.
Main.Start()

-- Roblox Shutdown Handling (Part 8): if the server is closing, finish
-- pending webhook requests and flush the queue before actually exiting,
-- rather than dropping whatever hadn't been sent yet.
pcall(function()
    game:BindToClose(function()
        Logger.Info("Main", "Server shutting down — flushing webhook queue")
        Utilities.SafeCall(Webhook.FlushQueue, "Main", 8)
        Logger.Info("Main", "Shutdown flush complete, exiting")
    end)
end)

return Main
